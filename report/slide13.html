Code Example: BLoC Testing

void main() {
  group('ConversationBloc Tests', () {
    late ConversationBloc conversationBloc;
    late MockConversationRepository mockRepository;
    late MockSpeechService mockSpeechService;
    
    setUp(() {
      mockRepository = MockConversationRepository();
      mockSpeechService = MockSpeechService();
      conversationBloc = ConversationBloc(
        conversationRepository: mockRepository,
        speechService: mockSpeechService,
      );
    });
    
    tearDown(() {
      conversationBloc.close();
    });
    
    test('initial state is correct', () {
      expect(conversationBloc.state, equals(ConversationInitial()));
    });
    
    blocTest(
      'emits [ConversationLoading, ConversationLoaded] when conversation is loaded successfully',
      build: () {
        when(mockRepository.getConversation(any))
            .thenAnswer((_) async => mockConversation);
        return conversationBloc;
      },
      act: (bloc) => bloc.add(LoadConversationEvent(conversationId: '123')),
      expect: () => [
        ConversationLoading(),
        ConversationLoaded(conversation: mockConversation),
      ],
    );
    
    // More tests for recording, sending messages, etc.
  });
}
