class ConversationBloc extends Bloc<ConversationEvent, ConversationState> {
  final ConversationRepository repository;
  final AudioService audioService;
  final UploadAudioUseCase uploadAudioUseCase;

  ConversationBloc({
    required this.repository,
    required this.audioService,
    required this.uploadAudioUseCase,
  }) : super(const ConversationInitial()) {
    on<CreateConversationEvent>(_onCreateConversation);
    on<SendSpeechMessageEvent>(_onSendSpeechMessage);
    on<GetMessageFeedbackEvent>(_onGetMessageFeedback);
    on<StartRecordingEvent>(_onStartRecording);
    on<StopRecordingEvent>(_onStopRecording);
    on<UploadAudioEvent>(_onUploadAudio);
    // Và các event handler khác...
  }

  // Xử lý SendSpeechMessageEvent
  Future<void> _onSendSpeechMessage(
      SendSpeechMessageEvent event, Emitter<ConversationState> emit) async {
    if (state.conversation == null) {
      emit(const ConversationError(message: 'No active conversation'));
      return;
    }

    emit(MessageSending(
      conversation: state.conversation!,
      audioId: event.audioId,
      transcription: state.transcription ?? '',
    ));

    final result = await repository.sendSpeechMessage(
      conversationId: event.conversationId,
      audioId: event.audioId,
    );

    emit(result.fold(
      (failure) => state.copyWith(
        isLoading: false,
        errorMessage: _mapFailureToMessage(failure),
      ),
      (messages) {
        return MessagesSent(
          conversation: state.conversation!.copyWith(
            messages: [
              ...state.conversation!.messages,
              messages.userMessage,
              messages.aiMessage,
            ],
          ),
          messages: messages,
        );
      },
    ));
  }
}
abstract class ConversationEvent extends Equatable {
  const ConversationEvent();

  @override
  List<Object?> get props => [];
}

// Tạo hội thoại mới
class CreateConversationEvent extends ConversationEvent {
  final String userRole;
  final String aiRole;
  final String situation;

  const CreateConversationEvent({
    required this.userRole,
    required this.aiRole,
    required this.situation,
  });
}

// Gửi tin nhắn âm thanh
class SendSpeechMessageEvent extends ConversationEvent {
  final String conversationId;
  final String audioId;

  const SendSpeechMessageEvent({
    required this.conversationId,
    required this.audioId,
  });
}

// Bắt đầu ghi âm
class StartRecordingEvent extends ConversationEvent {}

// Dừng ghi âm
class StopRecordingEvent extends ConversationEvent {
  final String filePath;
  final String? transcription;

  const StopRecordingEvent({
    required this.filePath,
    this.transcription,
  });
}
// Trạng thái ghi âm
enum RecordingState {
  // Không ghi âm
  idle,
  // Đang ghi âm
  recording,
  // Đang xử lý âm thanh
  processing,
  // Đã phiên âm xong
  transcribed
}

// Base class cho state
class ConversationState extends Equatable {
  final Conversation? conversation;
  final bool isLoading;
  final String? errorMessage;
  final RecordingState recordingState;
  final Feedback? activeFeedback;
  final String? transcription;
  final String? audioId;
  // ... nhiều thuộc tính khác
  
  const ConversationState({ ... });
  
  copyWith({ ... }) { ... }
}

// State khi đang gửi tin nhắn
class MessageSending extends ConversationState {
  const MessageSending({
    required Conversation conversation,
    required String audioId,
    required String transcription,
  }) : super(
    conversation: conversation,
    audioId: audioId,
    transcription: transcription,
    isLoading: true,
  );
}

// State khi đã gửi tin nhắn thành công
class MessagesSent extends ConversationState {
  const MessagesSent({
    required Conversation conversation,
    required ConversationMessages messages,
  }) : super(
    conversation: conversation,
    lastMessages: messages,
    recordingState: RecordingState.idle,
  );
}
